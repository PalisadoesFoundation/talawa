Future<QueryResult<Object?>> updatePledge(
  Map<String, dynamic> variables,
) async {
  try {
    final result = await _dbFunctions.gqlAuthMutation(
      _fundQueries.updatePledge(),
      variables: variables,
    );
    return result;
  } catch (e) {
    throw Exception('Failed to load pledges');
  }
}
