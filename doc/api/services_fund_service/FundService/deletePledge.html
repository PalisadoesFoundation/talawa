Future<QueryResult<Object?>> deletePledge(String pledgeId) async {
  try {
    final result = await _dbFunctions.gqlAuthMutation(
      _fundQueries.deletePledge(),
      variables: {
        'id': pledgeId,
      },
    );
    return result;
  } catch (e) {
    throw Exception('Failed to delete pledges');
  }
}
