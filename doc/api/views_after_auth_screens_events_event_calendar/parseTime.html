DateTime parseTime(String time) {
  try {
    return DateFormat('h:mm a', 'en_US').parse(time);
  } on FormatException {
    try {
      return DateFormat('Hms').parse(time);
    } catch (e) {
      throw Exception('Invalid time format: $time');
    }
  }
}
