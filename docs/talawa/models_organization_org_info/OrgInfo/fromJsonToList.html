List<OrgInfo> fromJsonToList(List<dynamic> json) {
  final List<OrgInfo> _orgList = [];
  json.forEach((element) {
    final OrgInfo org = OrgInfo.fromJson(element as Map<String, dynamic>);
    _orgList.add(org);
  });
  return _orgList;
}
