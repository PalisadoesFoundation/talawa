Implementation

Future<bool> isReachable({
  http.Client? client,
  String? uriString,
}) async {
  try {
    client ??= _client;
    await client
        .get(Uri.parse(uriString ?? graphqlConfig.httpLink.uri.toString()))
        .timeout(const Duration(seconds: 30));
    return true;
  } catch (e) {
    print('Timeout while checking reachability: $e');
    return false;
  }
}
